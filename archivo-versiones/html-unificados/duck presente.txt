### 🚀 2. La Experiencia "Zero-Setup" (Instalación en 1 Clic)

El verdadero "Super Regalo" es que Duck **no tenga que abrir la terminal** para instalar Python, pip, ni dependencias. Vamos a crear un script ejecutable que haga todo por él.

Crea estos dos archivos en la raíz del proyecto (`projeto-farol-duck/`):

#### Para Windows (`start.bat`)
```bat
@echo off
title 🦆 Projeto Farol | DuckOS
color 0A

echo ============================================================
echo    BEM-VINDO AO PROJETO FAROL (DuckOS)
echo    Instalando ambiente e iniciando servidor...
echo ============================================================
echo.

:: Verifica se o Python esta instalado
python --version >nul 2>&1
if %errorlevel% neq 0 (
    echo [ERRO] Python nao encontrado! Por favor, instale o Python em python.org
    pause
    exit /b
)

:: Cria ambiente virtual se nao existir
if not exist "venv" (
    echo Criando ambiente virtual...
    python -m venv venv
)

:: Ativa ambiente e instala dependencias
call venv\Scripts\activate
echo Instalando dependencias (isso so acontece na primeira vez)...
pip install -q -r requirements.txt

:: Inicia o Streamlit
echo.
echo Iniciando o Farol... O navegador abrira automaticamente.
echo Para parar, feche esta janela.
echo ============================================================
streamlit run app.py
```

#### Para Mac/Linux (`start.sh`)
```bash
#!/bin/bash
echo "============================================================"
echo "   BEM-VINDO AO PROJETO FAROL (DuckOS)"
echo "   Instalando ambiente e iniciando servidor..."
echo "============================================================"

# Verifica Python
if ! command -v python3 &> /dev/null; then
    echo "[ERRO] Python3 nao encontrado!"
    exit 1
fi

# Cria venv se nao existir
if [ ! -d "venv" ]; then
    echo "Criando ambiente virtual..."
    python3 -m venv venv
fi

# Ativa e instala
source venv/bin/activate
echo "Instalando dependencias..."
pip install -q -r requirements.txt

# Roda
echo "Iniciando o Farol..."
streamlit run app.py
```

**`core/llm_client.py`**
*(Adaptado para Google Gemini con manejo de errores y reintentos)*
```python
import google.generativeai as genai
import time
from typing import Optional
from google.api_core.exceptions import ResourceExhausted, GoogleAPICallError

class LLMClient:
    def __init__(self, api_key: str, model: str = "gemini-1.5-flash"):
        if not api_key:
            raise ValueError("API key do Google não pode estar vazia.")
        genai.configure(api_key=api_key)
        self.model_name = model

    def send_message(
        self,
        prompt: str,
        system_prompt: Optional[str] = None,
        max_tokens: int = 800,
        temperature: float = 0.7,
        retry_count: int = 3,
        retry_delay: float = 2.0
    ) -> str:
        
        model = genai.GenerativeModel(
            self.model_name,
            system_instruction=system_prompt
        )
        
        last_error = None
        for attempt in range(retry_count):
            try:
                response = model.generate_content(
                    prompt,
                    generation_config=genai.types.GenerationConfig(
                        temperature=temperature,
                        max_output_tokens=max_tokens
                    )
                )
                return response.text
            
            except ResourceExhausted as e:
                last_error = e
                wait = retry_delay * (2 ** attempt)
                time.sleep(wait)
            except GoogleAPICallError as e:
                last_error = e
                if attempt < retry_count - 1:
                    time.sleep(retry_delay)
                else:
                    raise Exception(f"Erro na API do Google: {e}")
            except Exception as e:
                last_error = e
                if attempt < retry_count - 1:
                    time.sleep(retry_delay)
                    
        raise Exception(f"Falha após {retry_count} tentativas. Último erro: {last_error}")
```

**`core/audio_analyzer.py`**
```python
import librosa
import numpy as np
import tempfile
import os
from typing import Any, Dict, Tuple, Union

AudioSource = Union[str, bytes, Any]

class AudioAnalyzer:
    def __init__(self, sample_rate: int = 22050, mono: bool = True):
        self.sample_rate = sample_rate
        self.mono = mono

    def load_audio(self, source: AudioSource) -> Tuple[np.ndarray, int]:
        if hasattr(source, "read"):
            return self._load_from_fileobj(source)
        return librosa.load(str(source), sr=self.sample_rate, mono=self.mono)

    def _load_from_fileobj(self, file_obj: Any) -> Tuple[np.ndarray, int]:
        file_obj.seek(0)
        audio_bytes = file_obj.read()
        suffix = ".wav"
        name = getattr(file_obj, "name", "")
        if name:
            ext = os.path.splitext(name)[1].lower()
            if ext in (".mp3", ".wav", ".flac", ".ogg", ".m4a"):
                suffix = ext
        return self._load_from_bytes(audio_bytes, suffix=suffix)

    def _load_from_bytes(self, audio_bytes: bytes, suffix: str = ".wav") -> Tuple[np.ndarray, int]:
        with tempfile.NamedTemporaryFile(suffix=suffix, delete=False) as tmp:
            tmp.write(audio_bytes)
            tmp_path = tmp.name
        try:
            y, sr = librosa.load(tmp_path, sr=self.sample_rate, mono=self.mono)
            return y, sr
        finally:
            if os.path.exists(tmp_path):
                os.unlink(tmp_path)

    def analyze(self, source: AudioSource) -> Dict[str, float]:
        y, sr = self.load_audio(source)
        centroid = librosa.feature.spectral_centroid(y=y, sr=sr)
        rms = librosa.feature.rms(y=y)
        rolloff = librosa.feature.spectral_rolloff(y=y, sr=sr, roll_percent=0.85)
        
        tempo = librosa.beat.beat_track(y=y, sr=sr)
        bpm = float(tempo[0]) if isinstance(tempo, tuple) else float(tempo)
        duration = librosa.get_duration(y=y, sr=sr)

        return {
            "spectral_centroid_mean": float(np.mean(centroid)),
            "rms_mean": float(np.mean(rms)),
            "spectral_rolloff_mean": float(np.mean(rolloff)),
            "tempo": bpm,
            "duration": float(duration),
            "sample_rate": sr,
        }

    def compare(self, mix_source: AudioSource, ref_source: AudioSource) -> Dict[str, Any]:
        mix_m = self.analyze(mix_source)
        ref_m = self.analyze(ref_source)

        centroid_diff = mix_m["spectral_centroid_mean"] - ref_m["spectral_centroid_mean"]
        rms_diff = mix_m["rms_mean"] - ref_m["rms_mean"]

        brightness = "mais brilhante" if centroid_diff > 0 else "mais escura"
        energy = "com mais energia" if rms_diff > 0 else "com menos energia"

        ref_centroid = ref_m["spectral_centroid_mean"] or 1.0
        ref_rms = ref_m["rms_mean"] or 1e-6

        centroid_pct = (abs(centroid_diff) / ref_centroid) * 100
        rms_pct = (abs(rms_diff) / ref_rms) * 100

        return {
            "mix": mix_m,
            "ref": ref_m,
            "comparison": {
                "brightness": brightness,
                "energy": energy,
                "centroid_pct_diff": float(centroid_pct),
                "rms_pct_diff": float(rms_pct),
                "brightness_description": f"Sua mix está {brightness} que a referência.",
                "energy_description": f"Sua mix está {energy} que a referência.",
            }
        }
```

```python
import yt_dlp
from typing import Dict
from config import VALID_URL_DOMAINS

class YouTubeExtractor:
    def __init__(self):
        self.ydl_opts = {"quiet": True, "no_warnings": True, "skip_download": True, "extract_flat": False}

    def is_valid_url(self, url: str) -> bool:
        if not url: return False
        return any(d in url.lower().strip() for d in VALID_URL_DOMAINS)

    def extract_metadata(self, url: str) -> Dict:
        if not self.is_valid_url(url):
            raise ValueError(f"URL inválida. Domínios suportados: {', '.join(VALID_URL_DOMAINS)}")
        try:
            with yt_dlp.YoutubeDL(self.ydl_opts) as ydl:
                info = ydl.extract_info(url, download=False)
                return {
                    "title": info.get("title", "Desconhecido"),
                    "description": (info.get("description") or "")[:1000],
                    "duration": info.get("duration", 0),
                    "uploader": info.get("uploader", "Desconhecido"),
                    "url": url,
                    "tags": info.get("tags", []) or [],
                }
        except Exception as e:
            raise ValueError(f"Erro ao acessar link: {e}")
```

```python
from typing import Any, Dict, Optional
from core.audio_analyzer import AudioAnalyzer
from core.llm_client import LLMClient
from prompts.templates import FAROL_SYSTEM, FAROL_USER

class FarolModule:
    def __init__(self, llm_client: LLMClient, audio_analyzer: Optional[AudioAnalyzer] = None):
        self.llm = llm_client
        self.analyzer = audio_analyzer or AudioAnalyzer()

    def run(self, mix_file: Any, ref_file: Any, user_intuition: str) -> Dict[str, Any]:
        if not user_intuition.strip():
            return self._error("Descreva sua intuição.")
        try:
            comparison = self.analyzer.compare(mix_file, ref_file)
            prompt = FAROL_USER.format(
                brightness_description=comparison["comparison"]["brightness_description"],
                energy_description=comparison["comparison"]["energy_description"],
                centroid_pct=comparison["comparison"]["centroid_pct_diff"],
                rms_pct=comparison["comparison"]["rms_pct_diff"],
                mix_bpm=comparison["mix"]["tempo"],
                ref_bpm=comparison["ref"]["tempo"],
                user_intuition=user_intuition.strip()
            )
            report = self.llm.send_message(prompt, system_prompt=FAROL_SYSTEM, max_tokens=800, temperature=0.6)
            return {"metrics": comparison, "report": report, "success": True, "error": None}
        except Exception as e:
            return self._error(f"Erro: {e}")

    @staticmethod
    def _error(msg: str) -> Dict[str, Any]:
        return {"metrics": None, "report": None, "success": False, "error": msg}
```

```python
from typing import Tuple
from config import VALID_AUDIO_EXTENSIONS, AUDIO_MAX_SIZE_MB

def validate_audio_file(file_obj) -> Tuple[bool, str | None]:
    if file_obj is None:
        return False, "Nenhum arquivo enviado."
    if hasattr(file_obj, "name"):
        ext = "." + file_obj.name.rsplit(".", 1)[-1].lower() if "." in file_obj.name else ""
        if ext not in VALID_AUDIO_EXTENSIONS:
            return False, f"Formato inválido. Use: {', '.join(VALID_AUDIO_EXTENSIONS)}"
    if hasattr(file_obj, "size") and file_obj.size:
        if file_obj.size > AUDIO_MAX_SIZE_MB * 1024 * 1024:
            return False, f"Arquivo > {AUDIO_MAX_SIZE_MB}MB."
    return True, None
```

**`core/llm_client.py`** *(Adaptado a Google Gemini con reintentos)*
```python
import google.generativeai as genai
import time
from typing import Optional
from google.api_core.exceptions import ResourceExhausted, GoogleAPICallError

class LLMClient:
    def __init__(self, api_key: str, model: str = "gemini-1.5-flash"):
        if not api_key:
            raise ValueError("API key de Google no puede estar vacía.")
        genai.configure(api_key=api_key)
        self.model_name = model

    def send_message(
        self,
        prompt: str,
        system_prompt: Optional[str] = None,
        max_tokens: int = 800,
        temperature: float = 0.7,
        retry_count: int = 3,
        retry_delay: float = 2.0
    ) -> str:
        model = genai.GenerativeModel(
            self.model_name,
            system_instruction=system_prompt
        )
        last_error = None
        for attempt in range(retry_count):
            try:
                response = model.generate_content(
                    prompt,
                    generation_config=genai.types.GenerationConfig(
                        temperature=temperature,
                        max_output_tokens=max_tokens
                    )
                )
                return response.text
            except ResourceExhausted as e:
                last_error = e
                wait = retry_delay * (2 ** attempt)
                time.sleep(wait)
            except GoogleAPICallError as e:
                last_error = e
                if attempt < retry_count - 1:
                    time.sleep(retry_delay)
                else:
                    raise Exception(f"Error en la API de Google: {e}")
            except Exception as e:
                last_error = e
                if attempt < retry_count - 1:
                    time.sleep(retry_delay)
        raise Exception(f"Fallo tras {retry_count} intentos. Último error: {last_error}")
```

**`core/audio_analyzer.py`**
```python
import librosa
import numpy as np
import tempfile
import os
from typing import Any, Dict, Tuple, Union

AudioSource = Union[str, bytes, Any]

class AudioAnalyzer:
    def __init__(self, sample_rate: int = 22050, mono: bool = True):
        self.sample_rate = sample_rate
        self.mono = mono

    def load_audio(self, source: AudioSource) -> Tuple[np.ndarray, int]:
        if hasattr(source, "read"):
            return self._load_from_fileobj(source)
        return librosa.load(str(source), sr=self.sample_rate, mono=self.mono)

    def _load_from_fileobj(self, file_obj: Any) -> Tuple[np.ndarray, int]:
        file_obj.seek(0)
        audio_bytes = file_obj.read()
        suffix = ".wav"
        name = getattr(file_obj, "name", "")
        if name:
            ext = os.path.splitext(name)[1].lower()
            if ext in (".mp3", ".wav", ".flac", ".ogg", ".m4a"):
                suffix = ext
        return self._load_from_bytes(audio_bytes, suffix=suffix)

    def _load_from_bytes(self, audio_bytes: bytes, suffix: str = ".wav") -> Tuple[np.ndarray, int]:
        with tempfile.NamedTemporaryFile(suffix=suffix, delete=False) as tmp:
            tmp.write(audio_bytes)
            tmp_path = tmp.name
        try:
            y, sr = librosa.load(tmp_path, sr=self.sample_rate, mono=self.mono)
            return y, sr
        finally:
            if os.path.exists(tmp_path):
                os.unlink(tmp_path)

    def analyze(self, source: AudioSource) -> Dict[str, float]:
        y, sr = self.load_audio(source)
        centroid = librosa.feature.spectral_centroid(y=y, sr=sr)
        rms = librosa.feature.rms(y=y)
        rolloff = librosa.feature.spectral_rolloff(y=y, sr=sr, roll_percent=0.85)
        tempo = librosa.beat.beat_track(y=y, sr=sr)
        bpm = float(tempo[0]) if isinstance(tempo, tuple) else float(tempo)
        duration = librosa.get_duration(y=y, sr=sr)
        return {
            "spectral_centroid_mean": float(np.mean(centroid)),
            "rms_mean": float(np.mean(rms)),
            "spectral_rolloff_mean": float(np.mean(rolloff)),
            "tempo": bpm,
            "duration": float(duration),
            "sample_rate": sr,
        }

    def compare(self, mix_source: AudioSource, ref_source: AudioSource) -> Dict[str, Any]:
        mix_m = self.analyze(mix_source)
        ref_m = self.analyze(ref_source)
        centroid_diff = mix_m["spectral_centroid_mean"] - ref_m["spectral_centroid_mean"]
        rms_diff = mix_m["rms_mean"] - ref_m["rms_mean"]
        brightness = "mais brilhante" if centroid_diff > 0 else "mais escura"
        energy = "com mais energia" if rms_diff > 0 else "com menos energia"
        ref_centroid = ref_m["spectral_centroid_mean"] or 1.0
        ref_rms = ref_m["rms_mean"] or 1e-6
        centroid_pct = (abs(centroid_diff) / ref_centroid) * 100
        rms_pct = (abs(rms_diff) / ref_rms) * 100
        return {
            "mix": mix_m,
            "ref": ref_m,
            "comparison": {
                "brightness": brightness,
                "energy": energy,
                "centroid_pct_diff": float(centroid_pct),
                "rms_pct_diff": float(rms_pct),
                "brightness_description": f"Sua mix está {brightness} que a referência.",
                "energy_description": f"Sua mix está {energy} que a referência.",
            }
        }
```

```python
import yt_dlp
from typing import Dict
from config import VALID_URL_DOMAINS

class YouTubeExtractor:
    def __init__(self):
        self.ydl_opts = {"quiet": True, "no_warnings": True, "skip_download": True, "extract_flat": False}

    def is_valid_url(self, url: str) -> bool:
        if not url: return False
        return any(d in url.lower().strip() for d in VALID_URL_DOMAINS)

    def extract_metadata(self, url: str) -> Dict:
        if not self.is_valid_url(url):
            raise ValueError(f"URL inválida. Dominios soportados: {', '.join(VALID_URL_DOMAINS)}")
        try:
            with yt_dlp.YoutubeDL(self.ydl_opts) as ydl:
                info = ydl.extract_info(url, download=False)
                return {
                    "title": info.get("title", "Desconhecido"),
                    "description": (info.get("description") or "")[:1000],
                    "duration": info.get("duration", 0),
                    "uploader": info.get("uploader", "Desconhecido"),
                    "url": url,
                    "tags": info.get("tags", []) or [],
                }
        except Exception as e:
            raise ValueError(f"Error al acceder al enlace: {e}")
```

```python
from typing import Any, Dict, Optional
from core.audio_analyzer import AudioAnalyzer
from core.llm_client import LLMClient
from prompts.templates import FAROL_SYSTEM, FAROL_USER

class FarolModule:
    def __init__(self, llm_client: LLMClient, audio_analyzer: Optional[AudioAnalyzer] = None):
        self.llm = llm_client
        self.analyzer = audio_analyzer or AudioAnalyzer()

    def run(self, mix_file: Any, ref_file: Any, user_intuition: str) -> Dict[str, Any]:
        if not user_intuition.strip():
            return self._error("Describe tu intuición.")
        try:
            comparison = self.analyzer.compare(mix_file, ref_file)
            prompt = FAROL_USER.format(
                brightness_description=comparison["comparison"]["brightness_description"],
                energy_description=comparison["comparison"]["energy_description"],
                centroid_pct=comparison["comparison"]["centroid_pct_diff"],
                rms_pct=comparison["comparison"]["rms_pct_diff"],
                mix_bpm=comparison["mix"]["tempo"],
                ref_bpm=comparison["ref"]["tempo"],
                user_intuition=user_intuition.strip()
            )
            report = self.llm.send_message(prompt, system_prompt=FAROL_SYSTEM, max_tokens=800, temperature=0.6)
            return {"metrics": comparison, "report": report, "success": True, "error": None}
        except Exception as e:
            return self._error(f"Error: {e}")

    @staticmethod
    def _error(msg: str) -> Dict[str, Any]:
        return {"metrics": None, "report": None, "success": False, "error": msg}
```

```python
from typing import Tuple
from config import VALID_AUDIO_EXTENSIONS, AUDIO_MAX_SIZE_MB

def validate_audio_file(file_obj) -> Tuple[bool, str | None]:
    if file_obj is None:
        return False, "Ningún archivo enviado."
    if hasattr(file_obj, "name"):
        ext = "." + file_obj.name.rsplit(".", 1)[-1].lower() if "." in file_obj.name else ""
        if ext not in VALID_AUDIO_EXTENSIONS:
            return False, f"Formato inválido. Usa: {', '.join(VALID_AUDIO_EXTENSIONS)}"
    if hasattr(file_obj, "size") and file_obj.size:
        if file_obj.size > AUDIO_MAX_SIZE_MB * 1024 * 1024:
            return False, f"Archivo > {AUDIO_MAX_SIZE_MB}MB."
    return True, None
```

### 📄 `core/llm_client.py`
```python
import google.generativeai as genai
import time
from typing import Optional
from google.api_core.exceptions import ResourceExhausted, GoogleAPICallError

class LLMClient:
    def __init__(self, api_key: str, model: str = "gemini-1.5-flash"):
        genai.configure(api_key=api_key)
        self.model_name = model

    def send_message(
        self,
        prompt: str,
        system_prompt: Optional[str] = None,
        max_tokens: int = 800,
        temperature: float = 0.7,
        retry_count: int = 3,
        retry_delay: float = 2.0
    ) -> str:
        model = genai.GenerativeModel(
            self.model_name,
            system_instruction=system_prompt
        )
        last_error = None
        for attempt in range(retry_count):
            try:
                response = model.generate_content(
                    prompt,
                    generation_config=genai.types.GenerationConfig(
                        temperature=temperature,
                        max_output_tokens=max_tokens
                    )
                )
                return response.text
            except ResourceExhausted as e:
                last_error = e
                time.sleep(retry_delay * (2 ** attempt))
            except GoogleAPICallError as e:
                last_error = e
                if attempt < retry_count - 1:
                    time.sleep(retry_delay)
                else:
                    raise Exception(f"Erro na API Google: {e}")
            except Exception as e:
                last_error = e
                if attempt < retry_count - 1:
                    time.sleep(retry_delay)
        raise Exception(f"Falha após {retry_count} tentativas. Último erro: {last_error}")
```

### 📄 `core/audio_analyzer.py`
```python
import librosa
import numpy as np
import tempfile, os
from typing import Any, Dict, Tuple, Union, List
import pyloudnorm as pyln

AudioSource = Union[str, bytes, Any]

class AudioAnalyzer:
    def __init__(self, sample_rate: int = 22050, mono: bool = False):
        self.sample_rate = sample_rate
        self.mono = mono  # mantener estéreo para medir anchura

    def load_audio(self, source: AudioSource) -> Tuple[np.ndarray, int]:
        if hasattr(source, "read"):
            return self._load_from_fileobj(source)
        return librosa.load(str(source), sr=self.sample_rate, mono=self.mono)

    def _load_from_fileobj(self, file_obj: Any) -> Tuple[np.ndarray, int]:
        file_obj.seek(0)
        audio_bytes = file_obj.read()
        suffix = ".wav"
        name = getattr(file_obj, "name", "")
        if name:
            ext = os.path.splitext(name)[1].lower()
            if ext in (".mp3", ".wav", ".flac", ".ogg", ".m4a"):
                suffix = ext
        return self._load_from_bytes(audio_bytes, suffix=suffix)

    def _load_from_bytes(self, audio_bytes: bytes, suffix: str = ".wav") -> Tuple[np.ndarray, int]:
        with tempfile.NamedTemporaryFile(suffix=suffix, delete=False) as tmp:
            tmp.write(audio_bytes)
            tmp_path = tmp.name
        try:
            y, sr = librosa.load(tmp_path, sr=self.sample_rate, mono=self.mono)
            return y, sr
        finally:
            if os.path.exists(tmp_path):
                os.unlink(tmp_path)

    def analyze(self, source: AudioSource) -> Dict[str, float]:
        y, sr = self.load_audio(source)
        # Convertir a mono para análisis si es necesario
        if y.ndim > 1:
            y_mono = librosa.to_mono(y)
        else:
            y_mono = y

        # Métricas espectrales
        centroid = librosa.feature.spectral_centroid(y=y_mono, sr=sr)
        rms = librosa.feature.rms(y=y_mono)
        rolloff = librosa.feature.spectral_rolloff(y=y_mono, sr=sr, roll_percent=0.85)
        tempo, _ = librosa.beat.beat_track(y=y_mono, sr=sr)
        duration = librosa.get_duration(y=y_mono, sr=sr)

        # LUFS
        meter = pyln.Meter(sr)
        # pyloudnorm espera un array float64 con forma (n,) o (n,2)
        if y.ndim == 1:
            lufs = meter.integrated_loudness(y)
        else:
            lufs = meter.integrated_loudness(y.T)  # (2, n) -> (n,2)

        # Bandas de frecuencia (aproximadas)
        stft = np.abs(librosa.stft(y_mono))
        freqs = librosa.fft_frequencies(sr=sr)
        low_mask = (freqs >= 20) & (freqs < 250)
        mid_mask = (freqs >= 250) & (freqs < 4000)
        high_mask = (freqs >= 4000)
        low_energy = float(np.mean(stft[low_mask, :]))
        mid_energy = float(np.mean(stft[mid_mask, :]))
        high_energy = float(np.mean(stft[high_mask, :]))
        total = low_energy + mid_energy + high_energy
        if total > 0:
            low_pct = (low_energy / total) * 100
            mid_pct = (mid_energy / total) * 100
            high_pct = (high_energy / total) * 100
        else:
            low_pct = mid_pct = high_pct = 0.0

        # Anchura estéreo (solo si es estéreo)
        stereo_width = 0.0
        if y.ndim > 1 and y.shape[0] == 2:
            left = y[0, :]
            right = y[1, :]
            # correlación entre canales
            corr = np.corrcoef(left, right)[0, 1] if len(left) > 1 else 1.0
            stereo_width = (1.0 - corr) * 100  # 0% = mono, 100% = totalmente descoordinado

        return {
            "spectral_centroid_mean": float(np.mean(centroid)),
            "rms_mean": float(np.mean(rms)),
            "spectral_rolloff_mean": float(np.mean(rolloff)),
            "tempo": float(tempo),
            "duration": duration,
            "sample_rate": sr,
            "lufs": float(lufs) if not np.isinf(lufs) else -24.0,
            "low_pct": low_pct,
            "mid_pct": mid_pct,
            "high_pct": high_pct,
            "stereo_width": stereo_width,
        }

    def compare(self, mix_source: AudioSource, ref_source: AudioSource) -> Dict[str, Any]:
        mix = self.analyze(mix_source)
        ref = self.analyze(ref_source)

        centroid_diff = mix["spectral_centroid_mean"] - ref["spectral_centroid_mean"]
        rms_diff = mix["rms_mean"] - ref["rms_mean"]
        lufs_diff = mix["lufs"] - ref["lufs"]

        brightness = "mais brilhante" if centroid_diff > 0 else "mais escura"
        energy = "com mais energia" if rms_diff > 0 else "com menos energia"

        ref_centroid = ref["spectral_centroid_mean"] or 1.0
        ref_rms = ref["rms_mean"] or 1e-6

        return {
            "mix": mix,
            "ref": ref,
            "differences": {
                "centroid_pct": (abs(centroid_diff) / ref_centroid) * 100,
                "rms_pct": (abs(rms_diff) / ref_rms) * 100,
                "lufs_diff": lufs_diff,
                "brightness_desc": f"Sua mix está {brightness} que a referência.",
                "energy_desc": f"Sua mix está {energy} que a referência.",
            }
        }
```

```python
import sqlite3
import json
from datetime import datetime
from config import DATABASE_PATH

class Database:
    def __init__(self, db_path=DATABASE_PATH):
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self.conn.row_factory = sqlite3.Row
        self._create_tables()

    def _create_tables(self):
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS projects (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                timestamp TEXT,
                module TEXT,
                title TEXT,
                metadata TEXT,
                report_md TEXT
            )
        """)
        self.conn.commit()

    def save_project(self, module: str, title: str, metadata: dict, report_md: str):
        self.conn.execute(
            "INSERT INTO projects (timestamp, module, title, metadata, report_md) VALUES (?,?,?,?,?)",
            (datetime.now().isoformat(), module, title, json.dumps(metadata), report_md)
        )
        self.conn.commit()

    def get_projects(self, module: str = None, limit: int = 50) -> list:
        if module:
            rows = self.conn.execute(
                "SELECT * FROM projects WHERE module = ? ORDER BY timestamp DESC LIMIT ?",
                (module, limit)
            ).fetchall()
        else:
            rows = self.conn.execute(
                "SELECT * FROM projects ORDER BY timestamp DESC LIMIT ?",
                (limit,)
            ).fetchall()
        return [dict(r) for r in rows]

    def get_project_by_id(self, project_id: int) -> dict:
        row = self.conn.execute("SELECT * FROM projects WHERE id = ?", (project_id,)).fetchone()
        return dict(row) if row else None

    def delete_project(self, project_id: int):
        self.conn.execute("DELETE FROM projects WHERE id = ?", (project_id,))
        self.conn.commit()

    def get_stats(self) -> dict:
        total = self.conn.execute("SELECT COUNT(*) FROM projects").fetchone()[0]
        by_module = {}
        for row in self.conn.execute("SELECT module, COUNT(*) as cnt FROM projects GROUP BY module").fetchall():
            by_module[row["module"]] = row["cnt"]
        return {"total": total, "by_module": by_module}
```

```python
import yt_dlp
from typing import Dict
from config import VALID_URL_DOMAINS

class YouTubeExtractor:
    def __init__(self):
        self.ydl_opts = {"quiet": True, "no_warnings": True, "skip_download": True, "extract_flat": False}

    def is_valid_url(self, url: str) -> bool:
        if not url:
            return False
        return any(d in url.lower().strip() for d in VALID_URL_DOMAINS)

    def extract_metadata(self, url: str) -> Dict:
        if not self.is_valid_url(url):
            raise ValueError(f"URL inválida. Domínios suportados: {', '.join(VALID_URL_DOMAINS)}")
        try:
            with yt_dlp.YoutubeDL(self.ydl_opts) as ydl:
                info = ydl.extract_info(url, download=False)
                # Si es una playlist, devolvemos lista de entradas
                if "entries" in info:
                    entries = []
                    for entry in info["entries"]:
                        entries.append({
                            "title": entry.get("title", "Desconhecido"),
                            "uploader": entry.get("uploader", "Desconhecido"),
                            "url": entry.get("webpage_url", url),
                            "description": (entry.get("description") or "")[:500],
                            "duration": entry.get("duration", 0),
                            "tags": entry.get("tags", []) or [],
                        })
                    return {
                        "type": "playlist",
                        "title": info.get("title", "Playlist"),
                        "entries": entries,
                        "url": url
                    }
                else:
                    return {
                        "type": "video",
                        "title": info.get("title", "Desconhecido"),
                        "description": (info.get("description") or "")[:1000],
                        "duration": info.get("duration", 0),
                        "uploader": info.get("uploader", "Desconhecido"),
                        "url": url,
                        "tags": info.get("tags", []) or [],
                    }
        except Exception as e:
            raise ValueError(f"Erro ao acessar link: {e}")
```

### 📄 `modules/farol.py`
```python
from typing import Any, Dict, Optional
from core.audio_analyzer import AudioAnalyzer
from core.llm_client import LLMClient
from core.database import Database
from prompts.templates import FAROL_SYSTEM, FAROL_USER

class FarolModule:
    def __init__(self, llm: LLMClient, analyzer: Optional[AudioAnalyzer] = None, db: Optional[Database] = None):
        self.llm = llm
        self.analyzer = analyzer or AudioAnalyzer()
        self.db = db

    def run(self, mix_file, ref_file, intuition: str) -> Dict[str, Any]:
        if not intuition.strip():
            return self._error("Descreva sua intuição.")
        try:
            comparison = self.analyzer.compare(mix_file, ref_file)
            mix = comparison["mix"]
            ref = comparison["ref"]
            diff = comparison["differences"]
            prompt = FAROL_USER.format(
                brightness_desc=diff["brightness_desc"],
                energy_desc=diff["energy_desc"],
                mix_lufs=mix["lufs"], ref_lufs=ref["lufs"],
                lufs_diff=diff["lufs_diff"],
                mix_low=mix["low_pct"], ref_low=ref["low_pct"],
                mix_mid=mix["mid_pct"], ref_mid=ref["mid_pct"],
                mix_high=mix["high_pct"], ref_high=ref["high_pct"],
                mix_width=mix["stereo_width"], ref_width=ref["stereo_width"],
                mix_bpm=mix["tempo"], ref_bpm=ref["tempo"],
                intuition=intuition.strip()
            )
            report = self.llm.send_message(prompt, system_prompt=FAROL_SYSTEM, max_tokens=900, temperature=0.65)
            title = f"Farol - {getattr(mix_file, 'name', 'audio')} vs {getattr(ref_file, 'name', 'ref')}"
            if self.db:
                self.db.save_project("Farol", title, {"metrics": comparison}, report)
            return {"success": True, "report": report, "comparison": comparison, "title": title}
        except Exception as e:
            return self._error(f"Erro: {e}")

    @staticmethod
    def _error(msg: str) -> Dict[str, Any]:
        return {"success": False, "error": msg}
```

```python
from typing import Any, Dict, Optional
from core.youtube_extractor import YouTubeExtractor
from core.llm_client import LLMClient
from core.database import Database
from prompts.templates import CARTOGRAFO_SYSTEM, CARTOGRAFO_USER

class CartografoModule:
    def __init__(self, llm: LLMClient, extractor: Optional[YouTubeExtractor] = None, db: Optional[Database] = None):
        self.llm = llm
        self.extractor = extractor or YouTubeExtractor()
        self.db = db

    def run(self, ref_link: str, obs: str = "") -> Dict[str, Any]:
        if not ref_link.strip():
            return self._error("Insira um link.")
        try:
            metadata = self.extractor.extract_metadata(ref_link.strip())
            if metadata["type"] == "playlist":
                # Tomamos la primera canción como referencia principal

                uploader = metadata["uploader"]

            if self.db:
                self.db.save_project("Cartógrafo", title, {"metadata": metadata}, blueprint)
            return {"success": True, "blueprint": blueprint, "title": title}
        except Exception as e:
            return self._error(str(e))

    def _error(msg: str) -> Dict[str, Any]:
```

```python
                artist=artist.strip(),
                license=license_type,
                extra_terms=extra_terms.strip() or "Nenhum"
            )
            package = self.llm.send_message(prompt, system_prompt=ENTREGA_SYSTEM, max_tokens=800, temperature=0.5)
            title = f"Entrega - {beat} ({artist})"
            if self.db:
                self.db.save_project("Entrega", title, {"beat": beat, "artist": artist, "license": license_type}, package)
            return {"success": True, "package": package, "title": title}
        except Exception as e:
            return self._error(f"Erro: {e}")

    @staticmethod
    def _error(msg: str) -> Dict[str, Any]:
        return {"success": False, "error": msg}
```

```python
from typing import Tuple
from config import VALID_AUDIO_EXTENSIONS, AUDIO_MAX_SIZE_MB

def validate_audio_file(file_obj) -> Tuple[bool, str]:
    if file_obj is None:
        return False, "Nenhum arquivo enviado."
    if hasattr(file_obj, "name"):
        ext = "." + file_obj.name.rsplit(".", 1)[-1].lower() if "." in file_obj.name else ""
        if ext not in VALID_AUDIO_EXTENSIONS:
            return False, f"Formato inválido. Use: {', '.join(VALID_AUDIO_EXTENSIONS)}"
    if hasattr(file_obj, "size") and file_obj.size:
        if file_obj.size > AUDIO_MAX_SIZE_MB * 1024 * 1024:
            return False, f"Arquivo excede {AUDIO_MAX_SIZE_MB}MB."
    return True, ""
```

### 📄 2. `core/audio_analyzer.py` (Nivel Mastering)
*Ahora calcula LUFS integrados, el dato más importante para un ingeniero de mezcla hoy en día.*
```python
import librosa
import numpy as np
import tempfile
import os
import pyloudnorm as pyln
from typing import Any, Dict, Tuple, Union

AudioSource = Union[str, bytes, Any]

class AudioAnalyzer:
    def __init__(self, sample_rate: int = 44100, mono: bool = True):
        # 44.1kHz es el estándar para LUFS y Mastering
        self.sample_rate = sample_rate
        self.mono = mono

    def load_audio(self, source: AudioSource) -> Tuple[np.ndarray, int]:
        if hasattr(source, "read"):
            return self._load_from_fileobj(source)
        return librosa.load(str(source), sr=self.sample_rate, mono=self.mono)

    def _load_from_fileobj(self, file_obj: Any) -> Tuple[np.ndarray, int]:
        file_obj.seek(0)
        audio_bytes = file_obj.read()
        suffix = ".wav"
        name = getattr(file_obj, "name", "")
        if name:
            ext = os.path.splitext(name)[1].lower()
            if ext in (".mp3", ".wav", ".flac", ".ogg", ".m4a"):
                suffix = ext
        return self._load_from_bytes(audio_bytes, suffix=suffix)

    def _load_from_bytes(self, audio_bytes: bytes, suffix: str = ".wav") -> Tuple[np.ndarray, int]:
        with tempfile.NamedTemporaryFile(suffix=suffix, delete=False) as tmp:
            tmp.write(audio_bytes)
            tmp_path = tmp.name
        try:
            y, sr = librosa.load(tmp_path, sr=self.sample_rate, mono=self.mono)
            return y, sr
        finally:
            if os.path.exists(tmp_path):
                os.unlink(tmp_path)

    def analyze(self, source: AudioSource) -> Dict[str, float]:
        y, sr = self.load_audio(source)
        
        # Espectro
        centroid = librosa.feature.spectral_centroid(y=y, sr=sr)
        rms = librosa.feature.rms(y=y)
        rolloff = librosa.feature.spectral_rolloff(y=y, sr=sr, roll_percent=0.85)
        
        # Tempo
        tempo = librosa.beat.beat_track(y=y, sr=sr)
        bpm = float(tempo[0]) if isinstance(tempo, tuple) else float(tempo)
        duration = librosa.get_duration(y=y, sr=sr)
        
        # 🚀 NUEVO: LUFS (Estándar Spotify/Apple Music)
        meter = pyln.Meter(sr)
        lufs = float(meter.integrated_loudness(y))
        # Normalizar LUFS si es infinito o nan
        if np.isinf(lufs) or np.isnan(lufs): lufs = -100.0 

        return {
            "spectral_centroid_mean": float(np.mean(centroid)),
            "rms_mean": float(np.mean(rms)),
            "spectral_rolloff_mean": float(np.mean(rolloff)),
            "tempo": bpm,
            "duration": float(duration),
            "lufs": lufs,
            "sample_rate": sr,
        }

    def compare(self, mix_source: AudioSource, ref_source: AudioSource) -> Dict[str, Any]:
        mix_m = self.analyze(mix_source)
        ref_m = self.analyze(ref_source)
        
        centroid_diff = mix_m["spectral_centroid_mean"] - ref_m["spectral_centroid_mean"]
        brightness = "mais brilhante" if centroid_diff > 0 else "mais escura"
        
        lufs_diff = mix_m["lufs"] - ref_m["lufs"]
        loudness_status = "mais alto (cuidado com clipping)" if lufs_diff > 1.0 else "mais baixo (precisa de gain)" if lufs_diff < -1.0 else "no volume ideal"

        return {
            "mix": mix_m,
            "ref": ref_m,
            "comparison": {
                "brightness": brightness,
                "lufs_diff": float(lufs_diff),
                "loudness_status": loudness_status,
                "brightness_description": f"Sua mix está {brightness} que a referência.",
                "lufs_description": f"LUFS da Mix: {mix_m['lufs']:.1f} | LUFS da Ref: {ref_m['lufs']:.1f}. Status: {loudness_status}.",
```

```python
import sqlite3
import os
from datetime import datetime

class DBManager:
    def __init__(self, db_name="farol_history.db"):
        self.db_name = db_name
        self._init_db()

    def _init_db(self):
        with sqlite3.connect(self.db_name) as conn:
            conn.execute('''
                CREATE TABLE IF NOT EXISTS reports (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    timestamp TEXT,
                    module TEXT,
                    identifier TEXT,
                    content TEXT
                )
            ''')

    def save_report(self, module: str, identifier: str, content: str):
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        with sqlite3.connect(self.db_name) as conn:
            conn.execute(
                "INSERT INTO reports (timestamp, module, identifier, content) VALUES (?, ?, ?, ?)",
                (timestamp, module, identifier, content)
            )

    def get_history(self, limit=10):
        with sqlite3.connect(self.db_name) as conn:
            cursor = conn.execute(
                "SELECT timestamp, module, identifier FROM reports ORDER BY id DESC LIMIT ?", 
                (limit,)
            )
            return cursor.fetchall()
```

### 📄 `core/llm_client.py`
```python
import google.generativeai as genai
import time
from typing import Optional
from google.api_core.exceptions import ResourceExhausted, GoogleAPICallError

class LLMClient:
    def __init__(self, api_key: str, model: str = "gemini-1.5-flash"):
        genai.configure(api_key=api_key)
        self.model_name = model

    def send_message(
        self,
        prompt: str,
        system_prompt: Optional[str] = None,
        max_tokens: int = 800,
        temperature: float = 0.7,
        retry_count: int = 3,
        retry_delay: float = 2.0
    ) -> str:
        model = genai.GenerativeModel(
            self.model_name,
            system_instruction=system_prompt
        )
        last_error = None
        for attempt in range(retry_count):
            try:
                response = model.generate_content(
                    prompt,
                    generation_config=genai.types.GenerationConfig(
                        temperature=temperature,
                        max_output_tokens=max_tokens
                    )
                )
                return response.text
            except ResourceExhausted as e:
                last_error = e
                time.sleep(retry_delay * (2 ** attempt))
            except GoogleAPICallError as e:
                last_error = e
                if attempt < retry_count - 1:
                    time.sleep(retry_delay)
                else:
                    raise Exception(f"Erro na API Google: {e}")
            except Exception as e:
                last_error = e
                if attempt < retry_count - 1:
                    time.sleep(retry_delay)
        raise Exception(f"Falha após {retry_count} tentativas. Último erro: {last_error}")
```

### 📄 `core/audio_analyzer.py`
*(Corregido el manejo de LUFS y la limitación de la anchura estéreo al 100%)*
```python
import librosa
import numpy as np
import tempfile, os
from typing import Any, Dict, Tuple, Union
import pyloudnorm as pyln

AudioSource = Union[str, bytes, Any]

class AudioAnalyzer:
    def __init__(self, sample_rate: int = 22050, mono: bool = False):
        self.sample_rate = sample_rate
        self.mono = mono  # manter estéreo para medir largura

    def load_audio(self, source: AudioSource) -> Tuple[np.ndarray, int]:
        if hasattr(source, "read"):
            return self._load_from_fileobj(source)
        return librosa.load(str(source), sr=self.sample_rate, mono=self.mono)

    def _load_from_fileobj(self, file_obj: Any) -> Tuple[np.ndarray, int]:
        file_obj.seek(0)
        audio_bytes = file_obj.read()
        suffix = ".wav"
        name = getattr(file_obj, "name", "")
        if name:
            ext = os.path.splitext(name)[1].lower()
            if ext in (".mp3", ".wav", ".flac", ".ogg", ".m4a"):
                suffix = ext
        return self._load_from_bytes(audio_bytes, suffix=suffix)

    def _load_from_bytes(self, audio_bytes: bytes, suffix: str = ".wav") -> Tuple[np.ndarray, int]:
        with tempfile.NamedTemporaryFile(suffix=suffix, delete=False) as tmp:
            tmp.write(audio_bytes)
            tmp_path = tmp.name
        try:
            y, sr = librosa.load(tmp_path, sr=self.sample_rate, mono=self.mono)
            return y, sr
        finally:
            if os.path.exists(tmp_path):
                os.unlink(tmp_path)

    def analyze(self, source: AudioSource) -> Dict[str, float]:
        y, sr = self.load_audio(source)
        if y.ndim > 1:
            y_mono = librosa.to_mono(y)
        else:
            y_mono = y

        centroid = librosa.feature.spectral_centroid(y=y_mono, sr=sr)
        rms = librosa.feature.rms(y=y_mono)
        rolloff = librosa.feature.spectral_rolloff(y=y_mono, sr=sr, roll_percent=0.85)
        tempo, _ = librosa.beat.beat_track(y=y_mono, sr=sr)
        duration = librosa.get_duration(y=y_mono, sr=sr)

        # LUFS
        meter = pyln.Meter(sr)
        if y.ndim == 1:
            lufs = meter.integrated_loudness(y)
        else:
            lufs = meter.integrated_loudness(y.T) 

        stft = np.abs(librosa.stft(y_mono))
        freqs = librosa.fft_frequencies(sr=sr)
        low_mask = (freqs >= 20) & (freqs < 250)
        mid_mask = (freqs >= 250) & (freqs < 4000)
        high_mask = (freqs >= 4000)
        
        low_energy = float(np.mean(stft[low_mask, :]))
        mid_energy = float(np.mean(stft[mid_mask, :]))
        high_energy = float(np.mean(stft[high_mask, :]))
        total = low_energy + mid_energy + high_energy
        
        if total > 0:
            low_pct = (low_energy / total) * 100
            mid_pct = (mid_energy / total) * 100
            high_pct = (high_energy / total) * 100
        else:
            low_pct = mid_pct = high_pct = 0.0

        stereo_width = 0.0
        if y.ndim > 1 and y.shape[0] == 2:
            left = y[0, :]
            right = y[1, :]
            corr = np.corrcoef(left, right)[0, 1] if len(left) > 1 else 1.0
            stereo_width = max(0.0, min(100.0, (1.0 - corr) * 100))

        return {
            "spectral_centroid_mean": float(np.mean(centroid)),
            "rms_mean": float(np.mean(rms)),
            "spectral_rolloff_mean": float(np.mean(rolloff)),
            "tempo": float(tempo),
            "duration": float(duration),
            "sample_rate": sr,
            "lufs": float(lufs) if not np.isinf(lufs) else -24.0,
            "low_pct": low_pct,
            "mid_pct": mid_pct,
            "high_pct": high_pct,
            "stereo_width": stereo_width,
        }

    def compare(self, mix_source: AudioSource, ref_source: AudioSource) -> Dict[str, Any]:
        mix = self.analyze(mix_source)
        ref = self.analyze(ref_source)

        centroid_diff = mix["spectral_centroid_mean"] - ref["spectral_centroid_mean"]
        rms_diff = mix["rms_mean"] - ref["rms_mean"]
        lufs_diff = mix["lufs"] - ref["lufs"]

        brightness = "mais brilhante" if centroid_diff > 0 else "mais escura"
        energy = "com mais energia" if rms_diff > 0 else "com menos energia"

        ref_centroid = ref["spectral_centroid_mean"] or 1.0
        ref_rms = ref["rms_mean"] or 1e-6

        return {
            "mix": mix,
            "ref": ref,
            "differences": {
                "centroid_pct": (abs(centroid_diff) / ref_centroid) * 100,
                "rms_pct": (abs(rms_diff) / ref_rms) * 100,
                "lufs_diff": lufs_diff,
                "brightness_desc": f"Sua mix está {brightness} que a referência.",
                "energy_desc": f"Sua mix está {energy} que a referência.",
            }
        }
```

```python
import sqlite3
import json
from datetime import datetime
from config import DATABASE_PATH

class Database:
    def __init__(self, db_path=DATABASE_PATH):
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self.conn.row_factory = sqlite3.Row
        self._create_tables()

    def _create_tables(self):
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS projects (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                timestamp TEXT,
                module TEXT,
                title TEXT,
                metadata TEXT,
                report_md TEXT
            )
        """)
        self.conn.commit()

    def save_project(self, module: str, title: str, metadata: dict, report_md: str):
        self.conn.execute(
            "INSERT INTO projects (timestamp, module, title, metadata, report_md) VALUES (?,?,?,?,?)",
            (datetime.now().isoformat(), module, title, json.dumps(metadata), report_md)
        )
        self.conn.commit()

    def get_projects(self, module: str = None, limit: int = 50) -> list:
        if module:
            rows = self.conn.execute(
                "SELECT * FROM projects WHERE module = ? ORDER BY timestamp DESC LIMIT ?",
                (module, limit)
            ).fetchall()
        else:
            rows = self.conn.execute(
                "SELECT * FROM projects ORDER BY timestamp DESC LIMIT ?",
                (limit,)
            ).fetchall()
        return [dict(r) for r in rows]

    def get_project_by_id(self, project_id: int) -> dict:
        row = self.conn.execute("SELECT * FROM projects WHERE id = ?", (project_id,)).fetchone()
        return dict(row) if row else None

    def delete_project(self, project_id: int):
        self.conn.execute("DELETE FROM projects WHERE id = ?", (project_id,))
        self.conn.commit()

    def get_stats(self) -> dict:
        total = self.conn.execute("SELECT COUNT(*) FROM projects").fetchone()[0]
        by_module = {}
        for row in self.conn.execute("SELECT module, COUNT(*) as cnt FROM projects GROUP BY module").fetchall():
            by_module[row["module"]] = row["cnt"]
        return {"total": total, "by_module": by_module}
```

### 📄 `core/youtube_extractor.py`

```python
import yt_dlp
from typing import Dict
from config import VALID_URL_DOMAINS

class YouTubeExtractor:
    def __init__(self):
        self.ydl_opts = {"quiet": True, "no_warnings": True, "skip_download": True, "extract_flat": False}

    def is_valid_url(self, url: str) -> bool:
        if not url:
            return False
        return any(d in url.lower().strip() for d in VALID_URL_DOMAINS)

    def extract_metadata(self, url: str) -> Dict:
        if not self.is_valid_url(url):
            raise ValueError(f"URL inválida. Domínios suportados: {', '.join(VALID_URL_DOMAINS)}")
        try:
            with yt_dlp.YoutubeDL(self.ydl_opts) as ydl:
                info = ydl.extract_info(url, download=False)
                if "entries" in info:
                    entries = []
                    for entry in info["entries"]:
                        entries.append({
                            "title": entry.get("title", "Desconhecido"),
                            "uploader": entry.get("uploader", "Desconhecido"),
                            "url": entry.get("webpage_url", url),
                            "description": (entry.get("description") or "")[:500],
                            "duration": entry.get("duration", 0),
                            "tags": entry.get("tags", []) or [],
                        })
                    return {
                        "type": "playlist",
                        "title": info.get("title", "Playlist"),
                        "entries": entries,
                        "url": url
                    }
                else:
                    return {
                        "type": "video",
                        "title": info.get("title", "Desconhecido"),
                        "description": (info.get("description") or "")[:1000],
                        "duration": info.get("duration", 0),
```

```python
from typing import Any, Dict, Optional
from core.audio_analyzer import AudioAnalyzer
from core.llm_client import LLMClient
from core.database import Database
from prompts.templates import FAROL_SYSTEM, FAROL_USER

class FarolModule:
    def __init__(self, llm: LLMClient, analyzer: Optional[AudioAnalyzer] = None, db: Optional[Database] = None):
        self.llm = llm
        self.analyzer = analyzer or AudioAnalyzer()
        self.db = db

    def run(self, mix_file, ref_file, intuition: str) -> Dict[str, Any]:
        if not intuition.strip():
            return self._error("Descreva sua intuição.")
        try:
            comparison = self.analyzer.compare(mix_file, ref_file)
            mix = comparison["mix"]
            ref = comparison["ref"]
            diff = comparison["differences"]
            prompt = FAROL_USER.format(
                brightness_desc=diff["brightness_desc"],
                energy_desc=diff["energy_desc"],
                mix_lufs=mix["lufs"], ref_lufs=ref["lufs"],
                lufs_diff=diff["lufs_diff"],
                mix_low=mix["low_pct"], ref_low=ref["low_pct"],
                mix_mid=mix["mid_pct"], ref_mid=ref["mid_pct"],
                mix_high=mix["high_pct"], ref_high=ref["high_pct"],
                mix_width=mix["stereo_width"], ref_width=ref["stereo_width"],
                mix_bpm=mix["tempo"], ref_bpm=ref["tempo"],
                intuition=intuition.strip()
            )
            report = self.llm.send_message(prompt, system_prompt=FAROL_SYSTEM, max_tokens=900, temperature=0.65)
            title = f"Farol - {getattr(mix_file, 'name', 'audio')} vs {getattr(ref_file, 'name', 'ref')}"
            if self.db:
                self.db.save_project("Farol", title, {"metrics": comparison}, report)
            return {"success": True, "report": report, "comparison": comparison, "title": title}
        except Exception as e:
            return self._error(f"Erro: {e}")

    @staticmethod
    def _error(msg: str) -> Dict[str, Any]:
        return {"success": False, "error": msg}
```

### 📄 `utils/helpers.py`
```python
from typing import Tuple
from config import VALID_AUDIO_EXTENSIONS, AUDIO_MAX_SIZE_MB

def validate_audio_file(file_obj) -> Tuple[bool, str]:
    if file_obj is None:
        return False, "Nenhum arquivo enviado."
    if hasattr(file_obj, "name"):
        ext = "." + file_obj.name.rsplit(".", 1)[-1].lower() if "." in file_obj.name else ""
        if ext not in VALID_AUDIO_EXTENSIONS:
            return False, f"Formato inválido. Use: {', '.join(VALID_AUDIO_EXTENSIONS)}"
    if hasattr(file_obj, "size") and file_obj.size:
        if file_obj.size > AUDIO_MAX_SIZE_MB * 1024 * 1024:
            return False, f"Arquivo excede {AUDIO_MAX_SIZE_MB}MB."
    return True, ""
```

```python
if modules_filter == "Todos":
    projects = db.get_projects(limit=100)
else:
    projects = db.get_projects(module=modules_filter, limit=100)

if not projects:
    st.info("Nenhum projeto encontrado.")
else:
    for proj in projects:
        with st.expander(f"{proj['timestamp'][:16]} — {proj['title']} ({proj['module']})"):
            st.markdown(proj["report_md"] if proj["report_md"] else "*Relatório vazio*")
            col1, col2 = st.columns(2)
            with col1:
                if proj["report_md"]:
                    st.download_button("📥 Baixar .md", proj["report_md"], file_name=f"{proj['title'][:30]}.md", key=f"down_{proj['id']}")
            with col2:
                if st.button("🗑️ Excluir", key=f"del_hist_{proj['id']}"):
                    db.delete_project(proj["id"])
                    st.rerun()
```
